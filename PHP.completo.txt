class Usuario {
  private $usuario;
  private $password;
  public function saludar() {
            echo 'Hola,' . $this->usuario . '¿Cómo estás?';
        }
        private function __construct() {
            $this->password = random_init(98123 , 56723);  
        }
        public function __destruct() {
            echo $this->usuario . 'Trabajo terminado';
        }
   public static function login ($nombre, $password){
      if ($this->usuario ==$nombre && $this->password ==$password)
           echo  'La contraseña es correcta';
    
       else{
           echo 'Esta contraseña no es correcta, inténtelo de nuevo';
   }
   
   }
   }
   
   class Administrador extends Usuario {  
    public function saludar() {
            echo 'Hola Horacio' . $this->usuario;
        }
        public function banear($usuario) {
            echo $usuario->usuario . " ha sido bloqueado";
        }
   }
